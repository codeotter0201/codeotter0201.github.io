# 先選取日盤時間
ohlcv = ohlcv.between_time('08:45', '13:45')
# 再用 index.date 進行聚合
ohlcv.groupby(ohlcv.index.date).agg({
    'open':'first',
    'high':'max',
    'low':'min',
    'close':'last',
    'volume':'sum',
}).reindex(ohlcv.index).ffill().tail()
